emptyColumns, staticColumns, totalColumns, rows :: Int
emptyColumns  = 4
staticColumns = 3
staticColsEnd = 1
totalColumns  = staticColumns + emptyColumns + staticColsEnd
rows          = 2^staticColumns


bothLangs :: OutputCapable m => String -> LangM m

getTask :: Gen (String, String, IO ([String],String))
getTask = do
    d@(x,y,z,_) <- taskData
    let [a,b,c] = sortOn fst [x,y,z]
    let (formulas,hints) = makeHintsAndFormula d
    let zipped = zip formulas hints
    let sorted = sortOn snd zipped
    let (formulasSorted,hintsSorted) = unzip sorted
    let lying = sort $ map (fromChar . fst) $ filter (\(_,x) -> x == False) [x,y,z]
    let solutionValues = map snd $ sortOn fst [x, y, z]
    let f = foldr1 (Binary And) formulasSorted
    let form = formify (Nothing :: Maybe (String, String, String))
                     [[single $ pack $ "Aussage von " ++ [fst a] ++ ":"],
                      [single $ pack $ "Aussage von " ++ [fst b] ++ ":"],
                      [single $ pack $ "Aussage von " ++ [fst c] ++ ":"]] $$>
               formify (Nothing :: Maybe (String))
                     [[single "Formel F="]] $$>
               tableForm ["A", "B", "C"] $$>
               formify (Nothing :: Maybe [Namen])
                     [[buttonsEnum Vertical "Wer lügt?" (pack . show @Namen)]]
    pure (show (hintsSorted,(lying,solutionValues,formulasSorted,f)), checkers False True, getFormData form)
  where
    fromChar 'A' = A
    fromChar 'B' = B
    fromChar 'C' = C

stringTuple :: String -> String -> String -> String -> String -> String
stringTuple s1 s2 s3 s4 s5 = '(':s1 ++ ',':s2 ++ ',':s3 ++ ',':s4 ++ ',':s5 ++ ")"


stringList :: [Maybe String] -> String
stringList xs = '[': intercalate "," (map stringMaybe xs) ++ "]"
  where
    stringMaybe Nothing = "Nothing"
    stringMaybe (Just s) = s


checkSyntax :: OutputCapable m => a -> FilePath -> Submission -> LangM m
checkSyntax _ _ (f1,f2,f3,f,Table xs, names) =
  parseDelayedWithAndThen
    parser
    complainAboutMissingParenthesesIfNotFailingOn
    allSymbolParser
    (checkSyntax' columns)
    $ delayed $ stringTuple (stringList headers) f1 f2 f3 f
  where
    (headers,columns) = unzip xs

checkSyntax' :: OutputCapable m => [[Maybe TruthValue]] -> ([Maybe (PropFormula Char)],PropFormula Char, PropFormula Char, PropFormula Char, PropFormula Char) -> LangM m
checkSyntax' columns (fs,f1,f2,f3,f) = do
  assertion (take staticColumns columns == #{startingTable}) $ bothLangs
    "Spalten der atomaren Formeln ergeben eine korrekte (aufsteigend geordnete) Wahrheitstafel?"
  pure ()


toTree :: PropFormula Char -> SynTree BinOp Char
toTree (Atomic c) = Leaf c
toTree (Neg p) = Not $ toTree p
toTree (Brackets p) = toTree p
toTree (Assoc op l r) = Binary op (toTree l) (toTree r)

truthValuesToBinaryToIndex :: [Bool] -> Int
truthValuesToBinaryToIndex = foldl (\\acc value -> acc * 2 + if value then 1 else 0) 0

generateTruthTable :: [Bool] -> [[Maybe TruthValue]]
generateTruthTable solutionValues =
  let solIndex = truthValuesToBinaryToIndex solutionValues
  in [[ if i == solIndex then
         Just (TruthValue True)
       else
         Just (TruthValue False)
     | i <- [0..7] ]]


checkSemantics :: OutputCapable m => (a,([Namen],[Bool],[SynTree BinOp Char],SynTree BinOp Char)) -> FilePath -> Submission -> Rated m
checkSemantics (_isSemanticEqual,(lying,solutionValues,formulas,solF)) _ (f1,f2,f3,f,Table xs,names) =
  withDelayedSucceeding
    (checkSemantics' formulas lying solutionValues columns solF names)
    parser
    $ delayed $ stringTuple (stringList headers) f1 f2 f3 f
  where
    (headers,columns) = unzip xs

checkSemantics' :: OutputCapable m => [SynTree BinOp Char] -> [Namen] -> [Bool] -> [[Maybe TruthValue]] -> SynTree BinOp Char
  -> [Namen] -> ([Maybe (PropFormula Char)],PropFormula Char, PropFormula Char, PropFormula Char, PropFormula Char) -> Rated m
checkSemantics' formulas lying solutionValues columns solF n (headers,f1,f2,f3,f) = do

  let correctParts = zipWith (\\fSol f -> isSemanticEqual fSol (toTree f)) formulas [f1,f2,f3]
  yesNo (and correctParts) $ bothLangs
    ("Die Aussagen sind korrekt übersetzt?" ++ show correctParts)

  let correctFormula = isSemanticEqual (toTree f) solF ||
                       (isSemanticEqual (toTree f) (toTree (foldr1 (Assoc And) (map Brackets [f1,f2,f3]))))
  yesNo correctFormula $ bothLangs
    ("Gesamtformel ist korrekt?" ++ simplestDisplay (toTree f) ++ " " ++ simplestDisplay solF)

  let correctLast = drop (staticColumns+emptyColumns) columns == generateTruthTable solutionValues
  yesNo correctLast $ bothLangs
    ("Spalte F der Wahrheitstafel korrekt?" ++ " 1 bei : " ++ show solutionValues)

  let correctNames = n == lying
  yesNo correctNames $ bothLangs
    ("Die Auswahl der Lügner ist korrekt?" ++ show lying ++ " " ++ show n)

  let points = ((if (and correctParts) then 0.4 else 0.0) +
                (if correctFormula then 0.2 else 0.0) +
                (if correctLast then 0.2 else 0.0) +
                (if correctNames then 0.2 else 0.0))
  pure points
|]

instance Parse Table where
  parseInput = do
    (headers,values) <- parseInput
    let allHeaders = map (Just . (:[])) "ABC" ++ headers ++ [Just "F"]
    pure $ Table $ zip allHeaders $ transpose $ chunksOf totalColumns values



parseSubmission :: String -> Either ParseError Submission
parseSubmission = useParser parseInput
